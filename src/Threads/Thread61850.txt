	}

	if ((sem1 = sem_open(SHM_SEM1_PATH, O_RDWR, 0644, 0)) == SEM_FAILED)

	{
		puts("<1>.semaphore is not exit");

	}

	{
		puts("<1>.semaphore is not exit");
		sleep(1);
		return NULL;
	}

	if ((mutex2_lock = sem_open(SHM_MUTEX2_PATH, O_RDWR, 0644, 0)) == SEM_FAILED)

     subscribeFromLcd();

{

		pcs_fault_flag[i]=0;
	}

	{
		printf(" thread_61850_write CREATE ERR!\n");

{
	// 	int portID;
	// int devID;
	// int typeID;
	// int pointID;
	data_info_t data_temp;
	int i = 0;


			

	for (i = 0; i < num; i++)
	{
		data_temp.map.data = data_temp.buf;
		data_temp.map.el_tag = el_tag;
		data_temp.map.sAddr = sAddr;
		switch (el_tag)
		{
		case _BOOL:
			data_temp.map.data = (char*)data;
			data_temp.map.data_size = 1;
			break;
		case _INT:
		case _UINT:
		case _FLOAT:

			memcpy(data_temp.map.data, (char*)data, 4);
			data_temp.map.data_size = 4;
			break;

		default:
			return;
		}

	}
}
